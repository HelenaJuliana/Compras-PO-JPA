CREATE TABLE CLIENTE (ID  SERIAL NOT NULL, CPF VARCHAR(255), EMAIL VARCHAR(255), ENDERECO VARCHAR(255), NOME VARCHAR(255), TELEFONE VARCHAR(255), VENDA_ID INTEGER, PRIMARY KEY (ID))
CREATE TABLE VENDA (ID  SERIAL NOT NULL, CPF VARCHAR(255), DATA VARCHAR(255), NOME VARCHAR(255), VALOR FLOAT, PRIMARY KEY (ID))
CREATE TABLE PRODUTO (ID  SERIAL NOT NULL, DESCRICAO VARCHAR(255), NOME_PRODUTO VARCHAR(255), PRECO FLOAT, QUANT_ESTOQ INTEGER, VENDA_ID INTEGER, PRIMARY KEY (ID))
CREATE TABLE FUNCIONARIO (ID  SERIAL NOT NULL, CPF VARCHAR(255), EMAIL VARCHAR(255), NASCIMENTO DATE, NOME VARCHAR(255), PRIMARY KEY (ID))
CREATE TABLE VENDA_CLIENTE (Venda_ID INTEGER NOT NULL, clientes_ID INTEGER NOT NULL, PRIMARY KEY (Venda_ID, clientes_ID))
CREATE TABLE VENDA_FUNCIONARIO (Venda_ID INTEGER NOT NULL, funcionarios_ID INTEGER NOT NULL, PRIMARY KEY (Venda_ID, funcionarios_ID))
CREATE TABLE PRODUTO_PRODUTO (Produto_ID INTEGER NOT NULL, produtos_ID INTEGER NOT NULL, PRIMARY KEY (Produto_ID, produtos_ID))

ALTER TABLE VENDA_CLIENTE ADD CONSTRAINT FK_VENDA_CLIENTE_Venda_ID FOREIGN KEY (Venda_ID) REFERENCES VENDA (ID)
ALTER TABLE VENDA_FUNCIONARIO ADD CONSTRAINT FK_VENDA_FUNCIONARIO_Venda_ID FOREIGN KEY (Venda_ID) REFERENCES VENDA (ID)
ALTER TABLE VENDA_FUNCIONARIO ADD CONSTRAINT FK_VENDA_FUNCIONARIO_funcionarios_ID FOREIGN KEY (funcionarios_ID) REFERENCES FUNCIONARIO (ID)
ALTER TABLE PRODUTO_PRODUTO ADD CONSTRAINT FK_PRODUTO_PRODUTO_Produto_ID FOREIGN KEY (Produto_ID) REFERENCES PRODUTO (ID)
